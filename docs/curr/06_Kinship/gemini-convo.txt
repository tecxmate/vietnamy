const CalculatorMode = () => {
  const [relType, setRelType] = useState('blood');
  const [gen, setGen] = useState('parent');
  const [side, setSide] = useState('paternal');
  const [relAge, setRelAge] = useState('older');
  const [gender, setGender] = useState('male');
  const [region, setRegion] = useState('north');

  const calculate = () => {
    if (relType === 'professional') return gender === 'male' ? 'Thầy' : 'Cô';
    if (relType === 'romantic') return gender === 'male' ? 'Anh' : 'Em';
    if (gen === 'grandparent') return gender === 'male' ? 'Ông' : 'Bà';
    if (gen === 'child') return 'Con / Cháu';
    if (gen === 'peer') return relAge === 'older' ? (gender === 'male' ? 'Anh' : 'Chị') : 'Em';
    
    // Parent Generation Logic
    if (side === 'paternal') {
      if (relAge === 'older') return gender === 'male' ? 'Bác' : (region === 'south' ? 'Cô' : 'Bác');
      return gender === 'male' ? 'Chú' : 'Cô';
    } else { // Maternal
      if (relAge === 'older') return region === 'north' ? 'Bác' : (gender === 'male' ? 'Cậu' : 'Dì');
      return gender === 'male' ? 'Cậu' : 'Dì';
    }
  };

  return (
    <div style={{ display: 'flex', flexDirection: 'column', gap: '15px' }}>
      <select value={relType} onChange={e => setRelType(e.target.value)} style={{ padding: '10px' }}>
        <option value="blood">Blood/Social Connection</option>
        <option value="romantic">Romantic Partner</option>
        <option value="professional">Teacher</option>
      </select>

      {relType === 'blood' && (
        <select value={gen} onChange={e => setGen(e.target.value)} style={{ padding: '10px' }}>
          <option value="grandparent">Grandparent (+2)</option>
          <option value="parent">Parent's Generation (+1)</option>
          <option value="peer">Peer (0)</option>
          <option value="child">Child (-1)</option>
        </select>
      )}

      {relType === 'blood' && gen === 'parent' && (
        <div style={{ display: 'flex', gap: '10px' }}>
          <select value={side} onChange={e => setSide(e.target.value)} style={{ padding: '10px', flex: 1 }}>
            <option value="paternal">Paternal (Nội)</option>
            <option value="maternal">Maternal (Ngoại)</option>
          </select>
          <select value={relAge} onChange={e => setRelAge(e.target.value)} style={{ padding: '10px', flex: 1 }}>
            <option value="older">Older</option>
            <option value="younger">Younger</option>
          </select>
        </div>
      )}

      {gen !== 'child' && (
        <select value={gender} onChange={e => setGender(e.target.value)} style={{ padding: '10px' }}>
          <option value="male">Male</option>
          <option value="female">Female</option>
        </select>
      )}

      {relType === 'blood' && gen === 'parent' && (
        <select value={region} onChange={e => setRegion(e.target.value)} style={{ padding: '10px' }}>
          <option value="north">Northern Dialect</option>
          <option value="south">Southern Dialect</option>
        </select>
      )}

      <div style={{ padding: '20px', backgroundColor: '#22c55e', color: 'white', textAlign: 'center', borderRadius: '8px' }}>
        <h2 style={{ margin: 0, fontSize: '2.5rem' }}>{calculate()}</h2>
      </div>
    </div>
  );
